Enter ride names separated by commas and click Sort.

from js import document def sort_rides(event=None): rides = document.getElementById("ride-input").value ride_list = [r.strip() for r in rides.split(",") if r.strip()] ride_list.sort() document.getElementById("sorted-rides").innerText = ", ".join(ride_list) document.getElementById("sort-rides-btn").addEventListener("click", sort_rides)
